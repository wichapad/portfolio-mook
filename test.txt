document.querySelectorAll(".mySwiper2").forEach((swiperEL) => {
    const id = swiperEL.dataset.id;
    const wrapper = swiperEL.querySelector(".carousel-wrapper");
    const images = carouselGroup[id];

    images.forEach((image) => {
      const slide = document.createElement("div");
      slide.className = "swiper-slide carousel-slide";
      slide.innerHTML = `<img src="${image}" />`;
      wrapper.appendChild(slide);
    });

    const isReverse = id === "2-2" || id === "2-4";
    new Swiper(swiperEL, {
      slidesPerView: 5,
      centeredSlides: true,
      spaceBetween: 16,
      direction: isReverse ? 'rtl' : "ltr",
      loop: true,
      autoplay: {
        delay: 0,
        disableOnInteraction: false,
        reverseDirection: isReverse, // ไปข้างหน้า
      },
      breakpoints: {
        320: { slidesPerView: 2 },
        640: { slidesPerView: 3 },
        992: { slidesPerView: 4 },
        1200: { slidesPerView: 5 },
      },

      speed: 3000, // ความเร็ว
    });
  });
